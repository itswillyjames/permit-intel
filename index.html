class Engine{

constructor(){

this.data=[];

this.lookback=7;

this.buyerFilter="all";

this.highOnly=false;

this.sources={

"Chicago":{url:"https://data.cityofchicago.org/resource/ydr8-5enu.json",date:"issue_date"},

"San Francisco":{url:"https://data.sfgov.org/resource/i98e-djp9.json",date:"issued_date"}

};

this.bind();

}

bind(){

document.getElementById("btn-run").onclick=()=>this.run();

document.getElementById("btn-json").onclick=()=>this.downloadJSON();

document.getElementById("btn-csv").onclick=()=>this.downloadCSV();

document.getElementById("btn-dossier").onclick=()=>this.generateDossier();

document.getElementById("buyer-filter").onchange=e=>{

this.buyerFilter=e.target.value;this.render();

};

document.getElementById("high-toggle").onchange=e=>{

this.highOnly=e.target.checked;this.render();

};

}

safeFloat(v){let n=parseFloat(v);return isNaN(n)?0:n;}

money(n){return new Intl.NumberFormat('en-US',{style:'currency',currency:'USD',maximumFractionDigits:0}).format(n||0)}

async fetchData(url){try{let r=await fetch(url);return await r.json()}catch(e){return []}}

async run(){

document.getElementById("loader").style.display="block";

this.data=[];

let cutoff=new Date();cutoff.setDate(cutoff.getDate()-this.lookback);

let raw=[];

for(let [city,cfg] of Object.entries(this.sources)){

let url=`${cfg.url}?$limit=2000&$order=${cfg.date} DESC`;

let rows=await this.fetchData(url);

rows.forEach(r=>{

let n=this.normalize(city,r);

if(!n)return;

let d=new Date(n.issued_date.split('T')[0]);

if(d>=cutoff)raw.push(n);

});

}

this.process(raw);

document.getElementById("btn-json").disabled=false;

document.getElementById("btn-csv").disabled=false;

document.getElementById("btn-dossier").disabled=false;

document.getElementById("loader").style.display="none";

this.render();

}

normalize(city,r){

let o={city,permit_id:"",permit_type:"",description:"",value:0,issued_date:"",address:"",developer_name:"",contractor_name:"",owner_name:""};

if(city==="Chicago"){

o.permit_id=r.permit_||"";

o.permit_type=r.permit_type||"";

o.description=(r.work_description||"").toLowerCase();

o.value=this.safeFloat(r.estimated_cost);

o.issued_date=r.issue_date||"";

o.address=[r.street_number,r.street_direction,r.street_name,r.suffix].filter(Boolean).join(" ").toUpperCase();

o.contractor_name=r.contact_1_name||"";

o.developer_name=r.contact_2_name||"";

o.owner_name=r.contact_3_name||"";

}

else{

o.permit_id=r.permit_number||"";

o.permit_type=r.permit_type_definition||"";

o.description=(r.description||"").toLowerCase();

o.value=this.safeFloat(r.estimated_cost);

o.issued_date=r.issued_date||"";

o.address=[r.street_number,r.street_name,r.street_suffix].filter(Boolean).join(" ").toUpperCase();

o.developer_name=r.owner_name||"";

}

if(!o.issued_date||!o.address)return null;

return o;

}

process(records){

let clusters={},devs={};

records.forEach(r=>{

clusters[r.address]=clusters[r.address]||{c:0,v:0};

clusters[r.address].c++;clusters[r.address].v+=r.value;

let d=r.developer_name||"UNKNOWN";

devs[d]=devs[d]||{cap:0,count:0};

devs[d].cap+=r.value;devs[d].count++;

});

this.data=records.map(r=>{

let desc=r.description,val=r.value;

let deal=0;

if(val>=1000000)deal=8;else if(val>=250000)deal=3;

if(desc.includes("new construction"))deal+=2;

if(desc.includes("addition"))deal+=2;

if(desc.includes("demolition"))deal+=2;

if(desc.includes("electrical"))deal+=1;

if(desc.includes("plumbing"))deal+=1;

if(desc.includes("roof"))deal+=1;

r.deal_score=Math.min(deal,12);

r.opportunity_class=r.deal_score>=10?"MAJOR":r.deal_score>=5?"MID":"MINOR";

r.cluster_size=clusters[r.address].c;

r.cluster_total_value=clusters[r.address].v;

let dev=devs[r.developer_name||"UNKNOWN"];

r.developer_total_capital=dev.cap;

r.developer_permit_count=dev.count;

r.capital_intensity=dev.cap>=5000000?"Institutional":dev.cap>=1000000?"Active Builder":"Small Operator";

let subs=[];

if(desc.includes("electrical"))subs.push("Electrical");

if(desc.includes("plumbing"))subs.push("Plumbing");

if(desc.includes("hvac"))subs.push("HVAC");

if(desc.includes("roof"))subs.push("Roofing");

if(desc.includes("structural"))subs.push("Structural");

if(desc.includes("demo"))subs.push("Demolition");

r.subtrades=subs;

let vertical="General";

if(/apartment|condo|multi-family|dwelling|units/.test(desc))vertical="Multi-Family";

else if(/retail|shop|store|restaurant/.test(desc))vertical="Retail";

else if(/office/.test(desc))vertical="Office";

else if(/warehouse|industrial/.test(desc))vertical="Industrial";

else if(/mixed-use/.test(desc))vertical="Mixed-Use";

r.primary_vertical=vertical;

let liq=0;

if(val>=1000000)liq+=3;

if(r.capital_intensity==="Institutional")liq+=2;

if(r.cluster_size>1)liq+=2;

if(vertical!=="General")liq+=2;

if(subs.length>0)liq+=1;

r.liquidity_score=Math.min(liq,10);

let fr=0;

if(!r.developer_name)fr+=2;

if(r.capital_intensity==="Small Operator")fr+=2;

if(val<100000)fr+=2;

if(subs.length===0)fr+=2;

if(vertical==="General")fr+=2;

r.friction_score=Math.min(fr,10);

let targets=[];

if(subs.includes("Electrical"))targets.push("Electrical Subcontractors");

if(val>1000000)targets.push("Commercial Lenders");

if(vertical==="Multi-Family")targets.push("Apartment Syndicators");

if(r.cluster_size>1)targets.push("Redevelopment Investors");

if(subs.length>0&&val<250000)targets.push("Trade-Level Resale");

if(!r.contractor_name)targets.push("GC Prospect");

r.buyer_targets=targets;

let arb=r.deal_score+(r.liquidity_score*1.5)-r.friction_score+(targets.length*1.2);

arb=Math.round(Math.max(0,Math.min(20,arb)));

r.arbitrage_score=arb;

r.fast_close=arb>=15?"High":arb>=10?"Medium":"Low";

r.immediate_close=(r.fast_close==="High"&&r.capital_intensity!=="Small Operator"&&val>=500000);

return r;

}).sort((a,b)=>b.arbitrage_score-a.arbitrage_score);

}

render(){

let d=this.data;

if(this.highOnly)d=d.filter(r=>r.fast_close==="High");

if(this.buyerFilter!=="all")d=d.filter(r=>r.buyer_targets.includes(this.buyerFilter));

document.getElementById("stat-signals").textContent=d.length;

document.getElementById("stat-value").textContent=this.money(d.reduce((s,r)=>s+r.value,0));

document.getElementById("stat-major").textContent=d.filter(r=>r.opportunity_class==="MAJOR").length;

document.getElementById("stat-close").textContent=d.filter(r=>r.immediate_close).length;

this.populate("grid-immediate",d.filter(r=>r.immediate_close));

this.populate("grid-top",d.slice(0,10));

this.populate("grid-fast",d.filter(r=>r.fast_close==="High"));

}

populate(id,list){

let el=document.getElementById(id);

el.innerHTML=list.map(r=>this.card(r)).join("")||"<div>No signals</div>";

}

card(r){

let dev=r.developer_name?(r.developer_name.length>18?r.developer_name.substring(0,18)+"...":r.developer_name):"N/A";

return `

<div class="card">

<div class="card-header">

<div class="title">${r.address}</div>

<div class="badge ${r.opportunity_class.toLowerCase()}">${r.opportunity_class}</div>

</div>

<div class="meta">

<div>City: ${r.city}</div>

<div class="money">${this.money(r.value)}</div>

<div>Dev: ${dev}</div>

<div>${r.capital_intensity}</div>

</div>

<div class="score-row">

<div>ARB ${r.arbitrage_score}</div>

<div>Deal ${r.deal_score}</div>

<div>Liq ${r.liquidity_score}</div>

<div>Fric ${r.friction_score}</div>

</div>

<div class="tags">

${r.fast_close==="High"?'<span class="tag fast">FAST</span>':''}

${r.buyer_targets.map(t=>`<span class="tag target">${t}</span>`).join("")}

</div>

<button onclick="engine.generateOutreach('${r.permit_id}')">Generate Angle</button>

</div>`;

}

generateOutreach(id){

let r=this.data.find(x=>x.permit_id===id);

if(!r)return;

let msg="";

if(r.capital_intensity==="Institutional"){

msg=`Institutional capital deploying on ${this.money(r.value)} at ${r.address}. Interested in early positioning before trade stacking?`;

}

else if(r.subtrades.includes("Electrical")){

msg=`Electrical scope flagged on ${this.money(r.value)} project at ${r.address}. Currently bidding similar work?`;

}

else if(r.cluster_size>1){

msg=`Multiple permits stacking at ${r.address}. Redevelopment momentum confirmed.`;

}

else{

msg=`Identified ${this.money(r.value)} opportunity at ${r.address}. Align with current targets?`;

}

alert(msg);

}

generateDossier(){

let top=this.data.slice(0,10);

let html=`<h1>Permit Intelligence Dossier</h1><p>${new Date().toLocaleString()}</p><hr>`;

top.forEach(r=>{

html+=`<h3>${r.address}</h3>

<p>City: ${r.city}</p>

<p>Value: ${this.money(r.value)}</p>

<p>Arbitrage Score: ${r.arbitrage_score}</p>

<p>Capital Intensity: ${r.capital_intensity}</p>

<p>Targets: ${r.buyer_targets.join(", ")}</p><hr>`;

});

let w=window.open("","_blank");

w.document.write(html);

w.document.close();

}

downloadJSON(){

let blob=new Blob([JSON.stringify(this.data,null,2)],{type:"application/json"});

let a=document.createElement("a");

a.href=URL.createObjectURL(blob);

a.download="permit_arbitrage_master.json";

a.click();

}

downloadCSV(){

if(!this.data.length)return;

let keys=Object.keys(this.data[0]);

let rows=[keys.join(",")];

this.data.forEach(r=>{

rows.push(keys.map(k=>`"${(Array.isArray(r[k])?r[k].join(";"):r[k]||"")}"`).join(","));

});

let blob=new Blob([rows.join("\n")],{type:"text/csv"});

let a=document.createElement("a");

a.href=URL.createObjectURL(blob);

a.download="permit_arbitrage_master.csv";

a.click();

}

}

let engine=new Engine();

</script>
